001    package com.odelia.grails.plugins.atmosphere;
002    
003    import org.codehaus.groovy.grails.commons.ArtefactHandlerAdapter;
004    
005    
006    /**
007     *
008     * @author BGoetzmann
009     */
010    public class AtmosphereHandlerArtefactHandler extends ArtefactHandlerAdapter {
011    
012        public static final String TYPE = "AtmosphereHandler";
013    
014        public AtmosphereHandlerArtefactHandler() {
015            super(TYPE, GrailsAtmosphereHandlerClass.class, DefaultGrailsAtmosphereHandlerClass.class, null);
016        }
017    
018        @Override
019        public boolean isArtefactClass(Class clazz) {
020            if (clazz == null) return false;
021    
022            return clazz.getName().endsWith(DefaultGrailsAtmosphereHandlerClass.ATMOSPHEREHANDLER);
023        }
024    
025    }
